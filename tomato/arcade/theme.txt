function gameMenu(Index:number) {
    EGP:delete("HeaderMask")
    
    let Frame = "Frame_"+Index
    let Game = "Game"+Index
    
    let FrameObj = EGP:egpobject(Elements:get(Frame))
    let GameObj = EGP:egpobject(Elements:get(Game))
    let MenuFrame = EGP:egpobject(Elements:get("MenuFrame"))
    
    FrameObj["radius", number] = 0
    GameObj["radius", number] = 0
    FrameObj["fidelity", number] = 1
    GameObj["fidelity", number] = 1
    MenuFrame["y", number] = 180
    
    transform(Frame, 0.5, vec2(445, 435))
    transform(Game, 0.5, vec2(445, 435))
    transform("Header", 0.5, vec2(155, 48))
    moveTo(Frame, 0.5, vec2(34, -115))
    moveTo(Game, 0.5, vec2(34, -115))
    moveTo("Header", 0.5, vec2(178, 17))
    box("MainMenu", table("x"=34.0, "y"=21.0, "w"=40.0, "h"=40.0, "r" = 255, "g" = 255, "b" = 255, "radius"=0, "material"="materials/zerochain/zerolib/ui/icon_back.png"))
    fadeIn("MainMenu", 2)
    
    loadingSpinner("TheSpinner", table("primary"=vec(255,255,255), "coords"=vec2(256,266), "scale"=0.5, "size"=8))
    
    switch (Index) {
        case 1,
            stopSpinner("TheSpinner")
            timer(1, function() {
                spaceInvaders(Frame)
            })
        break
        case 2,
        break
        case 3,
        break
        case 4,
        break
        case 5,
        break
        case 6,
        break
    }
}
